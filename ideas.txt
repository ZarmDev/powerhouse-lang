// if you want tools that you need, you import it. However, it's encouraged that you try to seperate files like below, where if you need functions in only a specific space, just use the using keyword because it's the only way to manage memory in powerhouse
import "servertools.ph"

server = new Server(8080)

server -> on("/login", "login.ph")
server -> on("/profile", "profile.ph")

// what the function would look like under the hood
fn on(x: str, route: str) -> void {
	user: str = _
	pass: str = _
	jwt* = _
	using "networking.ph" {
		connection* = accept_connection()
		body:str = connection->str
		user = find_user(body)
		pass = find_pass(body)
	}
	// once you use load_temp, you CANNOT run another load_temp inside of the using keyword. But, you can still use global variables/functions
	using "login.ph" {
		jwt* = login(user, pass)
	}
}

// start server at port 8080
server -> start("8080")
